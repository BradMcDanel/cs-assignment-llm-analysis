id:
- This column represents a unique identifier for each incident.
- Examples: 3, 4, 5

date:
- This column contains the date of the incident.
- Examples: 2015-01-02, 2015-01-02, 2015-01-03

threat_type:
- This column describes the type of threat perceived by the police.
- Examples: point, move, point

flee_status:
- This column indicates if the subject was fleeing and how.
- Examples: not, not, not

armed_with:
- This column describes what the subject was armed with.
- Examples: gun, gun, unarmed

city:
- This column contains the city where the incident occurred.
- Examples: Shelton, Aloha, Wichita

county:
- This column contains the county where the incident occurred.
- Examples: Mason, Washington, Sedgwick

state:
- This column contains the state where the incident occurred.
- Examples: WA, OR, KS

latitude:
- This column contains the latitude of the incident location.
- Examples: 47.246826, 45.4874214, 37.694766

longitude:
- This column contains the longitude of the incident location.
- Examples: -123.121592, -122.8916961, -97.280554

location_precision:
- Indicates the precision of the location data.
- Examples: not_available, not_available, not_available

name:
- This column contains the name of the person shot.
- Examples: Tim Elliot, Lewis Lee Lembke, John Paul Quintero

age:
- This column contains the age of the person shot.
- Examples: 53, 47, 23

gender:
- This column contains the gender of the person shot.
- Examples: male, male, male

race:
- This column contains the race of the person shot.
- Examples: A, W, H

race_source:
- This column contains the source of the race information.
- Examples: not_available, not_available, not_available

was_mental_illness_related:
- Indicates if mental illness was related to the incident.

body_camera:

agency_ids:
